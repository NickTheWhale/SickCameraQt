83 const inline CellStat calculateCellStatistics(int frameWidth, int frameHeight, const std::vector<uint16_t>& frame, QPoint topLeft, QPoint bottomRight)
84 {
85 auto stat = CellStat(0, 0, 0, 0);
86
87 // get the frame data contained by the cell
88 std::vector<uint16_t> cellData;
89 for (auto y = topLeft.y(); y < bottomRight.y(); ++y)
90 {
91 for (auto x = topLeft.x(); x < bottomRight.x(); ++x)
92 {
93 auto val = frame[y * frameWidth + x];
94 // we only care about values greater than 0
95 if (val > 0)
96 cellData.push_back(val);
97 }
98 }
99
100 if (cellData.size() > 0)
101 {
102 // min
103 const uint16_t min = *std::min_element(cellData.begin(), cellData.end());
104 // max
105 const uint16_t max = *std::max_element(cellData.begin(), cellData.end());
106 // mean
107 size_t sum = std::accumulate(std::begin(cellData), std::end(cellData), 0);
108 const size_t mean = sum / cellData.size();
109 // stdev
110 size_t accum = 0;
111 for (const auto& d : cellData)
112 {
113 accum += (d - mean) * (d - mean);
114 }
115 const size_t stdev = static_cast<size_t>(sqrt(accum / (cellData.size() - 1)));
116
117 stat.min = min;
118 stat.max = max;
119 stat.mean = mean;
120 stat.stdev = stdev;
121 }
122
123 return stat;
124 }

138 const inline unsigned long calculateFingerprint(int frameWidth, int frameHeight, const std::vector<uint16_t>& frame)
139 {
140 auto cStat = CellStat(0, 0, 0, 0);
141 for (auto row = 0; row < GRID_ROWS; ++row)
142 {
143 for (auto col = 0; col < GRID_COLS; ++col)
144 {
145 const auto topLeft = QPoint(col * (frameWidth / GRID_COLS), row * (frameHeight / GRID_ROWS));
146 const auto bottomRight = topLeft + QPoint(frameWidth / GRID_COLS, frameHeight / GRID_ROWS);
147 const auto stat = calculateCellStatistics(frameWidth, frameHeight, frame, topLeft, bottomRight);
148 cStat += stat;
149 }
150 }
151 return cStat.min + cStat.max + cStat.mean + cStat.stdev;
152 }
