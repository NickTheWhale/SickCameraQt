54 const inline CellStat calculateCellStatistics(int frameWidth, int frameHeight, const std::vector<uint16_t>& frame, QPoint topLeft, QPoint bottomRight)
55 {
56 auto stat = CellStat(0, 0, 0, 0);
57
58 std::vector<uint16_t> cellData;
59 for (auto y = topLeft.y(); y < bottomRight.y(); ++y)
60 {
61 for (auto x = topLeft.x(); x < bottomRight.x(); ++x)
62 {
63 auto val = frame[y * frameWidth + x];
64 if (val > 0)
65 cellData.push_back(val);
66 }
67 }
68
69 if (cellData.size() > 0)
70 {
71 // min
72 const uint16_t min = *std::min_element(cellData.begin(), cellData.end());
73 // max
74 const uint16_t max = *std::max_element(cellData.begin(), cellData.end());
75 // mean
76 size_t sum = std::accumulate(std::begin(cellData), std::end(cellData), 0);
77 const size_t mean = sum / cellData.size();
78 // stdev
79 size_t accum = 0;
80 for (const auto& d : cellData)
81 {
82 accum += (d - mean) * (d - mean);
83 }
84 const size_t stdev = static_cast<size_t>(sqrt(accum / (cellData.size() - 1)));
85
86 stat.min = min;
87 stat.max = max;
88 stat.mean = mean;
89 stat.stdev = stdev;
90 }
91
92 return stat;
93 }

95 const inline unsigned long calculateFingerprint(int frameWidth, int frameHeight, const std::vector<uint16_t>& frame)
96 {
97 auto cStat = CellStat(0, 0, 0, 0);
98 for (auto row = 0; row < GRID_ROWS; ++row)
99 {
100 for (auto col = 0; col < GRID_COLS; ++col)
101 {
102 const auto topLeft = QPoint(col * (frameWidth / GRID_COLS), row * (frameHeight / GRID_ROWS));
103 const auto bottomRight = topLeft + QPoint(frameWidth / GRID_COLS, frameHeight / GRID_ROWS);
104 const auto stat = calculateCellStatistics(frameWidth, frameHeight, frame, topLeft, bottomRight);
105 cStat += stat;
106 }
107 }
108 return cStat.min + cStat.max + cStat.mean + cStat.stdev;
109 }
